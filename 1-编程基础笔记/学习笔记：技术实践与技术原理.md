**3. 分支管理**

```shell
# 分支的增-查-改-删
# 新增分支
$ git branch new_branch_name
$ git push origin new_branch_name
$ git checkout new_branch_name
$ git checkout origin/master  #切换到远程主分支
#查看与删除远程分支
$ git branch -r 	#'-r' remote 查看远程分支
$ git branch -r -d origin/branch_name
$ git push origin :branch_name
# 查看分支信息与区别

```

**4. 修改与提交**

```shell
$ git add  #提交具体文件夹
$
# fork同步
$ git remote -v # 查看远程仓库
$ git remote add upstream  #添加上游仓库
$ git fetch upstream # 拉取远程上游到远程本地缓存
$ git checkout master #切换到主分支
$ git merge upstream/master #合并主分支
$ git push #推送修改到fork仓库
$ git status
```

```shell
$ make install

$ source .bashrc  #使设置生效

```
